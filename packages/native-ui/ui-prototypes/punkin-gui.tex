-- UI Component helpers
function rect(x, y, w, h, style)
    tex.sprint("\\node[" .. style .. "] at (" .. x .. "," .. y .. ") {};")
end

function label(x, y, text, style)
    tex.sprint("\\node[" .. style .. "] at (" .. x .. "," .. y .. ") {" .. text .. "};")
end

function button(x, y, w, h, text)
    tex.sprint("\\node[button, minimum width=" .. w .. "pt, minimum height=" .. h .. "pt] at (" .. x .. "," .. y .. ") {" .. text .. "};")
end

function card(x, y, w, h, style, content)
    tex.sprint("\\node[" .. style .. ", minimum width=" .. w .. "pt, minimum height=" .. h .. "pt, anchor=north west, align=left] at (" .. x .. "," .. y .. ") {" .. content .. "};")
end
